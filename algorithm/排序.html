    class Solution {
    public int[] sortArray(int[] nums) {
        int len= nums.length;
        for(int i = 0; i<len; i++){
                for(int j = i; j>0  && (nums[j] <nums[j-1]); j--){
                    // j>0 不等于0 , 因为nums[j] <nums[j-1]
                        exch(nums, j-1, j);        
                }
        }
        return nums;   
    }
    private void exch(int[] nums, int i, int j){
        int t =nums[i];
        nums[i] = nums[j];
        nums[j] =t;
    }   
}

    • 颜色分类

    给定一个包含红色、白色和蓝色，一共 n 个元素的数组，原地对它们进行排序，使得相同颜色的元素相邻，并按照红色、白色、蓝色顺序排列。

    此题中，我们使用整数 0、 1 和 2 分别表示红色、白色和蓝色。

    输入: [2,0,2,1,1,0]

    输出: [0,0,1,1,2,2]

    class Solution {
    public void sortColors(int[] nums) {
        int len = nums.length-1;
        int lo = 0, i = lo;
        //key i开始的位置， 与确定大小的元素对比时 从0开始
        //                 与第一个元素相比的时候(sort array);
            while(i <= len){
                if(nums[i] < 1){
                swap(nums,i++, lo++);
                }
                else if(nums[i] > 1){
                    swap(nums, i, len--);
                }
                else{
                    i++;
                }
            }
    }
    private void swap(int[]nums, int i , int j){
        int t  = nums[i];
        nums[i] = nums[j];
        nums[j] = t;
    }
}

    class Solution {
    public int[] sortArray(int[] nums) {   
        quick(nums, 0, nums.length-1);
        return nums;
    }
    public void quick(int[] nums, int lo, int hi){
        if(hi<=lo) return; 
        int lt = lo, i = lo+1, gt = hi;
        int c = nums[lo];
        while(i<=gt){
            if(nums[i] < c ){
                exch(nums, i++, lt++);
            }
            else if(nums[i]> c){
                exch(nums, gt--, i);
            }
            else {
                i++;
            }
        }
        quick (nums,lo, lt-1);
        quick(nums,gt+1, hi);
    }
    private void exch(int[] nums, int i, int j){
        int t =nums[i];
        nums[i] = nums[j];
        nums[j] =t;
    }   
}
